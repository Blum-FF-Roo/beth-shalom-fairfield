rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Hero slider images - only authenticated admin users can upload/delete
    match /hero-slider/{filename} {
      allow read: if true; // Anyone can read hero images (public website)
      allow write, delete: if request.auth != null; // Only authenticated users can upload/delete
    }
    
    // Additional rules for other potential folders
    match /avatars/{userId}/{filename} {
      allow read: if true;
      allow write, delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // Deny all other paths by default
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
